## Usage - Link Button

```jsx
/* Editor */
import { RicosEditor } from 'ricos-editor';
import { pluginLinkButton } from 'wix-rich-content-plugin-button';
import 'ricos-editor/dist/styles.min.css';
import 'wix-rich-content-plugin-button/dist/styles.min.css';
<RicosEditor plugins={[pluginLinkButton(LinkButtonConfig)]} />;

/* Viewer */
import { RicosViewer } from 'ricos-viewer';
import { pluginLinkButton } from 'wix-rich-content-plugin-button/dist/module.viewer';
import 'wix-rich-content-plugin-button/dist/styles.min.css';
<RicosViewer plugins={[pluginLinkButton(LinkButtonConfig)]} />;
```

## Link & Action Button Config API

### `palette`

Control Button palette colors

```jsx
palette?: string[];
```

Example value:
```jsx
palette: ['#FEFDFD', '#D5D4D4', '#ABCAFF', '#81B0FF', '#0261FF', '#0141AA'];
```

### `selectionBackgroundColor`

Control Button selection background color

```jsx
selectionBackgroundColor?: string;
```

Example value:

```jsx
selectionBackgroundColor: 'fuchsia';
```

### `selectionBorderColor`

Control Button selection border color

```jsx
selectionBorderColor?: string;
```

Example value:

```jsx
selectionBorderColor: '#FFF';
```

### `selectionTextColor`

Control Button selection text color

```jsx
selectionTextColor?: string;
```

Example value:

```jsx
selectionTextColor: '#FFF';
```

### `colors`

Control Button colors

```jsx
colors?: {
  color1?: string,
  color2?: string,
  color3?: string,
  ...
  `color${Number}?`: string,
}
```

Example value:

```jsx
colors = {
  color1: '#FEFDFD',
  color2: '#D5D4D4',
  color3: '#000000',
  color4: '#000000',
  color5: '#000000',
  color6: '#ABCAFF',
  color7: '#81B0FF',
  color8: '#0261FF',
  color9: '#0141AA',
  color10: '#012055',
}
```

### `onTextColorAdded`

```jsx
onTextColorAdded: (color: string) => string[]
```
A function that receives a color and returns user text colors.

### `onBackgroundColorAdded`

```jsx
onBackgroundColorAdded: (color: string) => string[]
```
A function that receives a color and returns user background colors.

### `onBorderColorAdded`

```jsx
onBorderColorAdded: (color: string) => string[]
```
A function that receives a color and returns user border colors.

### `getTextColors`

```jsx
getTextColors: () => string[]
```
A function that returns user text colors.

### `getBorderColors`

```jsx
getBorderColors: () => string[]
```
A function that returns user border colors.

### `getBackgroundColors`

```jsx
getBackgroundColors: () => string[]
```
A function that returns user background colors.

## Action Button Config API

### `onClick`

```jsx
onClick: () => void
```
A function that is triggered when clicking the button, it will trigger the `onClick` action.
