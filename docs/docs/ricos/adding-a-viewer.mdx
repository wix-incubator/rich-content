Now let's add some plugins to add ability of displaying plugin content.

```bash
npm i wix-rich-content-plugin-video wix-rich-content-plugin-divider
```

```jsx
import { WixRichContent } from "@wix/rich-content";
import { RichContentEditor } from "wix-rich-content-editor";

import { pluginVideo } from "wix-rich-content-plugin-video/dist/module.viewer.cjs";
import "wix-rich-content-plugin-video/dist/styles.min.css";

import { pluginDivider } from "wix-rich-content-plugin-divider/dist/module.viewer.cjs";
import "wix-rich-content-plugin-divider/dist/styles.min.css";

<RicosViewer
  initialState={contentWithPlugins}
  plugins={[pluginDivider(), pluginVideo()]}
```
