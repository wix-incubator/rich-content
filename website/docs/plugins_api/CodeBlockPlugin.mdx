import { pluginCodeBlock } from 'ricos/code-block/editor';
import { createContent } from 'ricos/content';
import { RicosEditorWithMobile } from '../../infra/editor/components/RicosEditorWithMobile.tsx';

Code Block plugin allows the user to show colored code snippets

<div className="card codeBlock">
  <RicosEditorWithMobile
    content={createContent('<Add a code block>')}
    placeholder={'Type here!'}
    plugins={[pluginCodeBlock()]}
  />
</div>
<br />

## Usage

```jsx
/* Editor */
import { RicosEditor } from 'ricos/editor';
import { pluginCodeBlock } from 'ricos/code-block/editor';

<RicosEditor plugins={[pluginCodeBlock()]} />;

/* Viewer */
import { RicosViewer } from 'ricos/viewer';
import { pluginCodeBlock } from 'ricos/code-block/viewer';

<RicosViewer plugins={[pluginCodeBlock()]} />;
```
